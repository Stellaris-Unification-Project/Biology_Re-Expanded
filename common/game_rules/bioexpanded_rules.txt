# Checks if a pop can be assembled
# This = Species
# Root = Planet
can_species_be_assembled = {
	is_sapient = yes
	OR = {
		if = {
			limit = {
				root = {
					has_planet_flag = bio_flag
				}
			}
			is_archetype = BIOLOGICAL
		}
		else_if = {
			limit = {
				root = {
					has_planet_flag = robo_flag
				}
			}
			is_archetype = ROBOT
		}
		is_archetype = MACHINE
	}
}
